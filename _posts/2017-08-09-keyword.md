{% highlight python %}
# avoid accidental passing of keyword arguments

a, b, *rest = range(10)
def f(a, b, *args, option=True):
  print(option)

# in python 2 this would have passed option=False, but not in python 3
f(a, b, False)
# in both python 2 and 3 this explictly passes option=False
f(a, b, option=False)
{% endhighlight %}
