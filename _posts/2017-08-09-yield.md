# easy way to use iterators in your generator

def weird_range():
  yield from range(5)
  yield from range(5)
  
for i in weird_range():
  print(i)

# in python 2 you had to manually iterate over the iterator

def python2_weird_range():
  for i in range(5):
    yield i
  for i in range(5):
    yield i
